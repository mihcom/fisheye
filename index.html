Sprite example 
var canvas = document.querySelector('#sprite-example'),
    options = {
        canvas: canvas, // where to render
        imageUrl: '/images/marc-jacobs.jpg', // sprite
        spriteImagesCount: 41 // how many images in sprite
        distortion: 2 // distortion level (default is 'max')
    };

new Fisheye(options);
